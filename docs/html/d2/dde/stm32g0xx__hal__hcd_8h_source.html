504__STATIC_INLINE uint16_t HCD_GET_CH_RX_CNT(HCD_TypeDef *Instance, uint16_t bChNum)
505{
506 uint32_t HostCoreSpeed;
507 uint32_t ep_reg = USB_DRD_GET_CHEP(Instance, bChNum);
508 __IO uint32_t count = 10U;
509
510 /* Get Host core Speed */
511 HostCoreSpeed = USB_GetHostSpeed(Instance);
512
513 /* Count depends on device LS */
514 if ((HostCoreSpeed == USB_DRD_SPEED_LS) || ((ep_reg & USB_CHEP_LSEP) == USB_CHEP_LSEP))
515 {
516 count = (70U * (HAL_RCC_GetHCLKFreq() / 1000000U)) / 100U;
517 }
518
519 if (count > 15U)
520 {
521 count = HCD_MAX(10U, (count - 15U));
522 }
523
524 /* WA: few cycles for RX PMA descriptor to update */
525 while (count > 0U)
526 {
527 count--;
528 }
529
530 return (uint16_t)USB_DRD_GET_CHEP_RX_CNT((Instance), (bChNum));
531}
