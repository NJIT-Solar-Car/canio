639/* PMA API Buffer Descriptor Management ------------------------------------------------------------*/
640/* Buffer Descriptor Table TXBD0/RXBD0 --- > TXBD7/RXBD7 8 possible descriptor
641* The buffer descriptor is located inside the packet buffer memory (USB_PMA_BUFF)
642* TXBD [Reserve |Countx| Address_Tx]
643* RXBD [BLSIEZ|NUM_Block |CounRx| Address_Rx] */
644
645/* Set TX Buffer Descriptor Address Field */
646#define USB_DRD_SET_CHEP_TX_ADDRESS(USBx, bEpChNum, wAddr) \
647 do { \
648 /* Reset old Address */ \
649 (USB_DRD_PMA_BUFF + (bEpChNum))->TXBD &= USB_PMA_TXBD_ADDMSK; \
650 \
651 /* Bit0 & Bit1 should be =0 PMA must be Word aligned */ \
652 (USB_DRD_PMA_BUFF + (bEpChNum))->TXBD |= (uint32_t)(((uint32_t)(wAddr) >> 2U) << 2U); \
653 } while(0) /* USB_DRD_SET_CHEP_TX_ADDRESS */

674 do { \
675 /* Divide PacketSize by 32 to calculate the Nb of Block32 */ \
676 (wNBlocks) =((uint32_t)(wCount) >> 5U); \
