1885__STATIC_INLINE void LL_USART_SetBaudRate(USART_TypeDef *USARTx, uint32_t PeriphClk, uint32_t PrescalerValue,
1886 uint32_t OverSampling,
1887 uint32_t BaudRate)
1888{
1889 uint32_t usartdiv;
1890 uint32_t brrtemp;
1891
1892 if (PrescalerValue > LL_USART_PRESCALER_DIV256)
1893 {
1894 /* Do not overstep the size of USART_PRESCALER_TAB */
1895 }
1896 else if (BaudRate == 0U)
1897 {
1898 /* Can Not divide per 0 */
1899 }
1900 else if (OverSampling == LL_USART_OVERSAMPLING_8)
1901 {
1902 usartdiv = (uint16_t)(__LL_USART_DIV_SAMPLING8(PeriphClk, (uint8_t)PrescalerValue, BaudRate));
1903 brrtemp = usartdiv & 0xFFF0U;
1904 brrtemp |= (uint16_t)((usartdiv & (uint16_t)0x000FU) >> 1U);
1905 USARTx->BRR = brrtemp;
1906 }
1907 else
1908 {
1909 USARTx->BRR = (uint16_t)(__LL_USART_DIV_SAMPLING16(PeriphClk, (uint8_t)PrescalerValue, BaudRate));
1910 }
1911}

1939__STATIC_INLINE uint32_t LL_USART_GetBaudRate(const USART_TypeDef *USARTx, uint32_t PeriphClk, uint32_t PrescalerValue,
1940 uint32_t OverSampling)
1941{
1942 uint32_t usartdiv;
1943 uint32_t brrresult = 0x0U;
1944 uint32_t periphclkpresc = (uint32_t)(PeriphClk / (USART_PRESCALER_TAB[(uint8_t)PrescalerValue]));
1945
1946 usartdiv = USARTx->BRR;
1947
1948 if (usartdiv == 0U)
1949 {
1950 /* Do not perform a division by 0 */
1951 }
1952 else if (OverSampling == LL_USART_OVERSAMPLING_8)
1953 {
1954 usartdiv = (uint16_t)((usartdiv & 0xFFF0U) | ((usartdiv & 0x0007U) << 1U)) ;
1955 if (usartdiv != 0U)
1956 {
1957 brrresult = (periphclkpresc * 2U) / usartdiv;
1958 }
1959 }
1960 else
1961 {
1962 if ((usartdiv & 0xFFFFU) != 0U)
1963 {
1964 brrresult = periphclkpresc / usartdiv;
1965 }
1966 }
1967 return (brrresult);
