47/* Internal mask for pair of comparators instances window mode: */
48/* To select into literals LL_COMP_WINDOWMODE_COMPx_INPUT_PLUS_COMMON */
49/* the relevant bits for: */
50/* (concatenation of multiple bits used in different registers) */
51/* - Comparator instance selected as master for window mode : register offset */
52/* - Window mode enable or disable: bit value */
53#define LL_COMP_WINDOWMODE_COMP_ODD_REGOFFSET_MASK (0x00000000UL) /* Register of COMP instance odd (COMP1_CSR, ...)
54 defined as reference register */
55#define LL_COMP_WINDOWMODE_COMP_EVEN_REGOFFSET_MASK (0x00000001UL) /* Register of COMP instance even (COMP2_CSR, ...)
56 offset vs register of COMP instance odd */
57#define LL_COMP_WINDOWMODE_COMPX_REGOFFSET_MASK (LL_COMP_WINDOWMODE_COMP_ODD_REGOFFSET_MASK \
58 | LL_COMP_WINDOWMODE_COMP_EVEN_REGOFFSET_MASK)

60#define LL_COMP_WINDOWOUTPUT_COMPX_SETTING_MASK (COMP_CSR_WINOUT)

367__STATIC_INLINE void LL_COMP_SetCommonWindowMode(COMP_Common_TypeDef *COMPxy_COMMON, uint32_t WindowMode)
368{
369 uint32_t window_mode_tmp = WindowMode;
370
371 /* Note: On this STM32 series, window mode can be set from any instance */
372 /* of the pair of comparator instances. */
373#if defined(COMP3)
374 /* Note: Exception for STM32G0 devices featuring ADC3 instance: in common */
375 /* group of COMP2 and COMP3, instances odd and even are inverted. */
376 /* Perform switch of parameter selected. */
377 if (COMPxy_COMMON == COMP23_COMMON)
378 {
379 if (WindowMode == LL_COMP_WINDOWMODE_COMP2_INPUT_PLUS_COMMON)
380 {
381 window_mode_tmp = LL_COMP_WINDOWMODE_COMP1_INPUT_PLUS_COMMON;
382 }
383 }
384#endif /* COMP3 */
385
386 __IO uint32_t *preg = __COMP_PTR_REG_OFFSET(COMPxy_COMMON->CSR_ODD,
387 (window_mode_tmp & LL_COMP_WINDOWMODE_COMPX_REGOFFSET_MASK));
388
389 /* Clear the potential previous setting of window mode */
390 __IO uint32_t *preg_clear = __COMP_PTR_REG_OFFSET(COMPxy_COMMON->CSR_ODD,
391 (~(window_mode_tmp & LL_COMP_WINDOWMODE_COMPX_REGOFFSET_MASK) & 0x1UL));
392 CLEAR_BIT(*preg_clear,
393 COMP_CSR_WINMODE
394 );
395
396 /* Set window mode */
397 MODIFY_REG(*preg,
398 COMP_CSR_WINMODE,
399 (window_mode_tmp & LL_COMP_WINDOWMODE_COMPX_SETTING_MASK)
400 );
401}

417__STATIC_INLINE uint32_t LL_COMP_GetCommonWindowMode(const COMP_Common_TypeDef *COMPxy_COMMON)
418{
419 /* Note: On this STM32 series, window mode can be set from any instance */
420 /* of the pair of comparator instances. */
421#if defined(COMP3)
422 uint32_t window_mode_comp_odd;
423 uint32_t window_mode_comp_even;
424
425 /* Note: Exception for STM32G0 devices featuring ADC3 instance: in common */
426 /* group of COMP2 and COMP3, instances odd and even are inverted. */
427 /* Perform switch of parameter selected. */
428 if (COMPxy_COMMON == COMP23_COMMON)
429 {
430 window_mode_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
431 window_mode_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINMODE);
432 }
433 else
434 {
435 window_mode_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINMODE);
436 window_mode_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
437 }
438#else
439 const uint32_t window_mode_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINMODE);
440 const uint32_t window_mode_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
441#endif /* COMP3 */
442 return (uint32_t)(window_mode_comp_odd
443 | window_mode_comp_even
444 | ((window_mode_comp_even >> LL_COMP_WINDOWMODE_BITOFFSET_POS)
445 * LL_COMP_WINDOWMODE_COMP_EVEN_REGOFFSET_MASK)
446 );
447}

466__STATIC_INLINE void LL_COMP_SetCommonWindowOutput(COMP_Common_TypeDef *COMPxy_COMMON, uint32_t WindowOutput)
467{
468 uint32_t window_output_tmp = WindowOutput;
469
470#if defined(COMP3)
471 /* Note: Exception for STM32G0 devices featuring ADC3 instance: in common */
472 /* group of COMP2 and COMP3, instances odd and even are inverted. */
473 /* Perform switch of parameter selected. */
474 if (COMPxy_COMMON == COMP23_COMMON)
475 {
476 if (WindowOutput == LL_COMP_WINDOWOUTPUT_COMP2)
477 {
478 window_output_tmp = LL_COMP_WINDOWOUTPUT_COMP3;
479 }
480 else if (WindowOutput == LL_COMP_WINDOWOUTPUT_COMP3)
481 {
482 window_output_tmp = LL_COMP_WINDOWOUTPUT_COMP2;
483 }
484 else
485 {
486 /* No modification of other parameters */
487 }
488 }
489#endif /* COMP3 */
490
491 __IO uint32_t *preg = __COMP_PTR_REG_OFFSET(COMPxy_COMMON->CSR_ODD, (window_output_tmp & LL_COMP_WINDOWMODE_COMPX_REGOFFSET_MASK));
492
493 /* Clear the potential previous setting of window output on the relevant comparator instance */
494 /* (clear bit of window output unless specific case of setting of comparator both output selected) */
495 __IO uint32_t *preg_clear = __COMP_PTR_REG_OFFSET(COMPxy_COMMON->CSR_ODD, (~(window_output_tmp & LL_COMP_WINDOWMODE_COMPX_REGOFFSET_MASK) & 0x1UL));
496 MODIFY_REG(*preg_clear,
497 COMP_CSR_WINOUT,
498 ((window_output_tmp & LL_COMP_WINDOWOUTPUT_BOTH_SETTING_MASK) >> LL_COMP_WINDOWOUTPUT_BOTH_POS_VS_WINDOW)
499 );
500
501 /* Set window output */
502 MODIFY_REG(*preg,
503 COMP_CSR_WINOUT,
504 (window_output_tmp & LL_COMP_WINDOWOUTPUT_COMPX_SETTING_MASK)
505 );
506}

524__STATIC_INLINE uint32_t LL_COMP_GetCommonWindowOutput(const COMP_Common_TypeDef *COMPxy_COMMON)
525{
526#if defined(COMP3)
527 uint32_t window_output_comp_odd;
528 uint32_t window_output_comp_even;
529
530 /* Note: Exception for STM32G0 devices featuring ADC3 instance: in common */
531 /* group of COMP2 and COMP3, instances odd and even are inverted. */
532 /* Perform switch of parameter selected. */
533 if (COMPxy_COMMON == COMP23_COMMON)
534 {
535 window_output_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINOUT);
536 window_output_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINOUT);
537 }
538 else
539 {
540 window_output_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINOUT);
541 window_output_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINOUT);
542 }
543#else
544 const uint32_t window_output_comp_odd = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_ODD, COMP_CSR_WINOUT);
545 const uint32_t window_output_comp_even = (uint32_t)READ_BIT(COMPxy_COMMON->CSR_EVEN, COMP_CSR_WINOUT);
546#endif /* COMP3 */
547
548 /* Construct value corresponding to LL_COMP_WINDOWOUTPUT_xxx */
549 return (uint32_t)(window_output_comp_odd
550 | window_output_comp_even
551 | ((window_output_comp_even >> COMP_CSR_WINOUT_Pos) * LL_COMP_WINDOWMODE_COMP_EVEN_REGOFFSET_MASK)
552 | (window_output_comp_odd + window_output_comp_even));
553}
